class Solution:
    def findKthNumber(self, n: int, k: int) -> int:
        curr=0
        def count(p,n,k):
            nonlocal curr
            p*=10
            for i in range(0,10):
                if p+i<=n: 
                    curr+=1
                    if curr==k:return p+i
                    res=count(p+i,n,k)
                    if curr==k:return res
            return 0

        for i in range(1,10): 
            if i<=n: 
                curr+=1
                if curr==k: return i
                res=count(i,n,k)
                if curr==k: return res

        return res
```

class Solution:
    def findKthNumber(self, n: int, k: int) -> int:
        def count(n,node1,node2):
            steps=0
            while node1<=n:
                steps+=min(n+1,node2)-node1
                node1*=10
                node2*=10
            return steps
        
        curr=1
        k-=1
        while k>0:
            steps=count(n,curr,curr+1)
            if steps<=k:
                k-=steps
                curr+=1
            else:
                k-=1
                curr*=10
        return curr
```
